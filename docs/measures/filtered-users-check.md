VAR _users_list =
	SUMMARIZE(
		'dim_Admin_OS',
		'dim_Admin_OS'[FULL_NAME],
		'dim_Admin_OS'[ORDER_NUM_2]
	)
VAR _res =
	CONCATENATEX (
		_users_list,
		'dim_Admin_OS'[FULL_NAME],
		", " & UNICHAR(10),
		'dim_Admin_OS'[ORDER_NUM_2],
		ASC
	)
RETURN
	_res
