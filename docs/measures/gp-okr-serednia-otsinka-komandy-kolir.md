VAR _res = IF(ISBLANK([GP.ОКР.Середня оцінка команди.Значення]), -1, [GP.ОКР.Середня оцінка команди.Значення])

VAR _color = 
SWITCH(
    TRUE(),
    _res >= 101, "Суперзелений",
    _res >= 91, "Зелений",

    _res >= 50, "Жовтий",
    _res >= 25, "Жовто-червоний",
    _res >= 0, "Червоний",
    "Неоціненно")

RETURN  _color
